resp = nsfg.ReadFemResp()
pmf = thinkstats2.Pmf(resp.numkdhh, label='actual')

def biaspmf(pmf, label):
    new_pmf = pmf.Copy(label=label)
    
    for x, p in pmf.Items():
        new_pmf.Mult(x,x)
        
        new_pmf.Normalize()
    return new_pmf
    
biased_pmf = biaspmf(pmf, 'observed')
    
thinkplot.PrePlot(2)
thinkplot.Pmfs([pmf, biased_pmf])
